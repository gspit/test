units           real
atom_style      full
boundary        p p p

# Force field settings
pair_style      lj/cut 10.0
bond_style      harmonic
angle_style     harmonic
dihedral_style  hybrid charmm fourier
improper_style  cvff

read_data       lammps.data

minimize 1e-6 1e-6 2000 20000

# =====================================================
# Dump trajectory for relaxation only
# =====================================================

dump 1 all custom 5000 pef.xyz id type x y z
dump_modify 1 sort id

timestep 1.0

# ---------------- 升温 ----------------
velocity all create 300.0 928374 dist gaussian mom yes rot yes

fix HEAT all npt temp 300.0 800.0 100.0 iso 1.0 1.0 1000.0
thermo_style custom step temp press pe etotal enthalpy vol density
thermo 1000
run 500000
unfix HEAT

# ---------------- 800K 驰豫 ----------------
fix RELAX all npt temp 800.0 800.0 100.0 iso 1.0 1.0 1000.0
run 1000000
unfix RELAX

# ---------------- 降温 ----------------
fix COOL all npt temp 800.0 300.0 100.0 iso 1.0 1.0 1000.0
run 500000
unfix COOL

# 保存结构
write_data npt_relaxed.data

# 关键修正：关闭dump，才能reset timestep
undump 1

# =====================================================
# 熔点扫描开始
# =====================================================

velocity all create 300.0 12345 dist gaussian mom yes rot yes

fix EQ all npt temp 300.0 300.0 100.0 iso 1.0 1.0 1000.0
thermo_style custom step temp press enthalpy vol
thermo 1000
run 100000
unfix EQ

shell echo "# T(K)    Enthalpy(kcal/mol)    Volume(A^3)" > H_T.dat

# -------- Coarse Scan --------

variable T equal 300
variable T_end equal 900
variable dT equal 20

label coarse_loop
if "${T} > ${T_end}" then "jump SELF coarse_done"

reset_timestep 0
velocity all create ${T} 45678 dist gaussian mom yes rot yes

fix NPT all npt temp ${T} ${T} 100.0 iso 1.0 1.0 1000.0
run 100000

variable H equal enthalpy
variable V equal vol
print "${T} ${H} ${V}" append H_T.dat

unfix NPT

variable T equal ${T}+${dT}
jump SELF coarse_loop

label coarse_done

# -------- Fine Scan --------

variable T equal 300
variable T_end equal 750
variable dT equal 5

label fine_loop
if "${T} > ${T_end}" then "jump SELF end_all"

reset_timestep 0
velocity all create ${T} 78901 dist gaussian mom yes rot yes

fix NPT all npt temp ${T} ${T} 100.0 iso 1.0 1.0 1000.0
run 150000

variable H equal enthalpy
variable V equal vol
print "${T} ${H} ${V}" append H_T.dat

unfix NPT

variable T equal ${T}+${dT}
jump SELF fine_loop

label end_all
